        function handleCalculateClick() {
            if (!calculateBtn) return;

            calculateBtn.disabled = true;
            calculateBtn.innerHTML = 'Calculating...';
            calculateBtn.setAttribute('aria-busy', 'true');
            downloadBtn.disabled = true;

            setTimeout(() => {
                const success = calculateImpact();
                calculateBtn.innerHTML = 'Calculate Impact';
                calculateBtn.disabled = false;
                calculateBtn.removeAttribute('aria-busy');

                if (success) {
                    downloadBtn.disabled = false;
                }

        function calculateImpact() {
            const industry = document.getElementById('industry').value;
            const companySize = document.getElementById('company-size').value;
            const revenueInput = document.getElementById('current-revenue');
            const revenueValue = revenueInput.value.trim();
            const revenue = Number(revenueValue);
            const useCase = document.getElementById('ai-use-case').value;
            const scope = document.getElementById('implementation-scope').value;
            const currentEfficiency = document.getElementById('current-efficiency').value;
            const techReadiness = document.getElementById('tech-readiness').value;
            const changeReadiness = document.getElementById('change-readiness').value;

            if (!Number.isFinite(revenue) || revenueValue === '') {
                revenueInput.classList.add('input-error');
                showError('Please enter a valid annual revenue (minimum $100,000)');
                calculatedResults = null;
                return false;
            }

            if (revenue < 100000) {
                revenueInput.classList.add('input-error');
                showError('Please enter a valid annual revenue (minimum $100,000)');
                calculatedResults = null;
                return false;
            }

            revenueInput.classList.remove('input-error');

            // Activate result opacity
            activateResults();

            // Base calculations with industry context
            const industryData = industryMultipliers[industry];
            const useCaseData = useCaseMultipliers[useCase];
            const scopeData = scopeMultipliers[scope];

            if (!industryData || !useCaseData || !scopeData) {
                calculatedResults = null;
                showError('Unable to calculate results with the current selections.');
                return false;
            }

            // Company size factors
            const sizeFactors = {
                'startup': { baseCost: 50000, efficiency: 1.2, complexity: 0.8 },
                'small': { baseCost: 150000, efficiency: 1.1, complexity: 0.9 },
                'medium': { baseCost: 400000, efficiency: 1.0, complexity: 1.0 },
                'large': { baseCost: 800000, efficiency: 0.9, complexity: 1.2 }
            };

            const sizeData = sizeFactors[companySize];

            // Efficiency baseline (what percentage of revenue could be optimized)
            let efficiencyBaseline = 0.08; // 8% baseline
            if (currentEfficiency === 'low') efficiencyBaseline = 0.12;
            if (currentEfficiency === 'medium') efficiencyBaseline = 0.08;
            if (currentEfficiency === 'high') efficiencyBaseline = 0.04;

            // Technology readiness impact
            let techMultiplier = 1.0;
            if (techReadiness === 'basic') techMultiplier = 0.7;
            if (techReadiness === 'intermediate') techMultiplier = 0.9;
            if (techReadiness === 'advanced') techMultiplier = 1.1;

            // Change readiness impact
            let changeMultiplier = 1.0;
            if (changeReadiness === 'resistant') changeMultiplier = 0.6;
            if (changeReadiness === 'neutral') changeMultiplier = 0.8;
            if (changeReadiness === 'adaptable') changeMultiplier = 1.0;

            // Calculate realistic annual savings
            const potentialSavings = revenue * efficiencyBaseline;
            const adjustedSavings = potentialSavings * 
                industryData.efficiency * 
                useCaseData.savings * 
                scopeData.savings * 
                sizeData.efficiency * 
                techMultiplier * 
                changeMultiplier * 
                0.7; // Conservative factor

            // Calculate implementation cost
            const baseCost = sizeData.baseCost * scopeData.cost;
            const complexityCost = baseCost * industryData.complexity * useCaseData.complexity * sizeData.complexity;
            const techCost = techReadiness === 'basic' ? baseCost * 0.4 : (techReadiness === 'intermediate' ? baseCost * 0.2 : 0);
            const changeCost = changeReadiness === 'resistant' ? baseCost * 0.3 : (changeReadiness === 'neutral' ? baseCost * 0.15 : 0);
            
            const totalImplementationCost = baseCost + complexityCost + techCost + changeCost;

            // Calculate 3-year ROI
            const yearOneRealization = adjustedSavings * 0.3;
            const yearTwoRealization = adjustedSavings * 0.7;
            const yearThreeRealization = adjustedSavings * 1.0;
            const threeYearSavings = yearOneRealization + yearTwoRealization + yearThreeRealization;
            const threeYearROI = ((threeYearSavings - totalImplementationCost) / totalImplementationCost) * 100;

            // Calculate payback period
            let paybackMonths = 0;
            let cumulativeSavings = 0;
            let monthlySavings = adjustedSavings / 12;
            
            for (let month = 1; month <= 60; month++) {
                if (month <= 12) {
                    cumulativeSavings += monthlySavings * 0.3;
                } else if (month <= 24) {
                    cumulativeSavings += monthlySavings * 0.7;
                } else {
                    cumulativeSavings += monthlySavings;
                }
                
                if (cumulativeSavings >= totalImplementationCost) {
                    paybackMonths = month;
                    break;
                }
            }

            // Calculate confidence level
            let confidenceScore = 50;
            
            if (['technology', 'marketing', 'retail'].includes(industry)) confidenceScore += 15;
            if (['manufacturing', 'logistics'].includes(industry)) confidenceScore += 10;
            if (['healthcare', 'financial'].includes(industry)) confidenceScore += 5;
            
            if (['automation', 'content', 'customer-service'].includes(useCase)) confidenceScore += 15;
            if (['analytics', 'optimization'].includes(useCase)) confidenceScore += 10;
            if (['predictive', 'personalization'].includes(useCase)) confidenceScore += 5;
            
            if (techReadiness === 'advanced') confidenceScore += 10;
            if (techReadiness === 'intermediate') confidenceScore += 5;
            if (techReadiness === 'basic') confidenceScore -= 10;
            
            if (changeReadiness === 'adaptable') confidenceScore += 10;
            if (changeReadiness === 'neutral') confidenceScore += 0;
            if (changeReadiness === 'resistant') confidenceScore -= 15;
            
            if (scope === 'pilot') confidenceScore += 10;
            if (scope === 'enterprise') confidenceScore -= 5;

            confidenceScore = Math.max(20, Math.min(85, confidenceScore));

            // Store results for PDF generation
            calculatedResults = {
                industry,
                companySize,
                revenue,
                useCase,
                scope,
                currentEfficiency,
                techReadiness,
                changeReadiness,
                adjustedSavings,
                totalImplementationCost,
                threeYearROI,
                paybackMonths,
                confidenceScore
            };

            // Update display
            updateResults(adjustedSavings, totalImplementationCost, threeYearROI, paybackMonths, confidenceScore);
            return true;
        }

        function updateResults(savings, cost, roi, payback, confidence) {
            document.getElementById('annual-savings').textContent = formatCurrency(savings);
            document.getElementById('implementation-cost').textContent = formatCurrency(cost);
            document.getElementById('roi-percentage').textContent = Math.round(roi) + '%';
            document.getElementById('payback-period').textContent = payback > 0 ? payback + ' months' : '60+ months';

            // ROI category
            let roiCategory = '';
            if (roi >= 200) roiCategory = 'Exceptional Return';
            else if (roi >= 100) roiCategory = 'Strong Return';
            else if (roi >= 50) roiCategory = 'Good Return';
            else if (roi >= 20) roiCategory = 'Moderate Return';
            else roiCategory = 'Low Return';
            
            document.getElementById('roi-category').textContent = roiCategory;

            // Confidence indicator with animation
            const confidenceDot = document.getElementById('confidence-dot');
            const confidenceText = document.getElementById('confidence-text');
            
            confidenceDot.classList.add('animate');
            setTimeout(() => confidenceDot.classList.remove('animate'), 300);
            
            if (confidence >= 70) {
                confidenceDot.className = 'confidence-dot confidence-high animate';
                confidenceText.textContent = `High Confidence (${confidence}%)`;
            } else if (confidence >= 50) {
                confidenceDot.className = 'confidence-dot confidence-medium animate';
                confidenceText.textContent = `Medium Confidence (${confidence}%)`;
            } else {
                confidenceDot.className = 'confidence-dot confidence-low animate';
                confidenceText.textContent = `Low Confidence (${confidence}%)`;
            }
        }

        function formatCurrency(amount) {
            if (typeof amount !== 'number' || !Number.isFinite(amount)) {
                return '$0';
            }

            const safeAmount = amount >= 0 ? Math.floor(amount) : Math.ceil(amount);
            return currencyFormatter.format(safeAmount);
        }

        function showError(message) {
            alert(message);
        }

        function requestConsultation() {
            if (!calculatedResults) {
                showError('Please calculate results before requesting a consultation.');
                return;
            }

            const confidenceSummary = document.getElementById('confidence-text').textContent;
            const results = {
                savings: formatCurrency(calculatedResults.adjustedSavings),
                cost: formatCurrency(calculatedResults.totalImplementationCost),
                roi: Math.round(calculatedResults.threeYearROI) + '%',

                confidence: confidenceSummary
            };
            
            localStorage.setItem('aiCalculatorResults', JSON.stringify(results));
            window.open('https://jservo.com/consultation', '_blank');
        }
